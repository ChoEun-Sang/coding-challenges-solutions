`## <a href='문제 링크'>코딩테스트 문제</a>`

```javascript
function solution(bandage, health, attacks) {
  const [t, x, y] = bandage; // 붕대 감기 기술 정보 추출
  let currentHealth = health; // 현재 체력 초기화
  const maxHealth = health; // 최대 체력 설정
  let currentTime = 0; // 현재 시간 초기화
  let healStreak = 0; // 연속 성공 시간 초기화

  for (let i = 0; i < attacks.length; i++) {
    const [attackTime, damage] = attacks[i];

    // 현재 시간부터 공격 시간 전까지 체력 회복
    while (currentTime < attackTime) {
      if (currentHealth <= 0) {
        return -1; // 캐릭터 사망
      }
      if (healStreak < t) {
        currentHealth = Math.min(currentHealth + x, maxHealth);
        healStreak += 1;
      } else {
        currentHealth = Math.min(currentHealth + y, maxHealth);
        healStreak = 0;
      }
      currentTime += 1;
    }

    // 공격으로 체력 감소
    currentHealth -= damage;
    healStreak = 0; // 공격 받으면 연속 성공 초기화
    currentTime += 1; // 다음 시간으로 이동

    if (currentHealth <= 0) {
      return -1; // 캐릭터 사망
    }
  }

  // 마지막 공격 이후 체력 회복
  while (healStreak < t) {
    if (currentHealth <= 0) {
      return -1; // 캐릭터 사망
    }
    currentHealth = Math.min(currentHealth + x, maxHealth);
    healStreak += 1;
  }

  return currentHealth > 0 ? currentHealth : -1;
}
```
